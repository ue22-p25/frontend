a hidden box that can be turned on from the button once the second snippet is run
this page has jQuery loaded
open it in an external window,
and run the following code in the console 

///////////////

// using selectors to apply treatments
$("button.continue").html("Next Step...")

///////////////

// set event handlers
$("#button-container button").on(
    "click",
     function( event ) {
        $("#banner-message").show();
});
$("button.continue").html("I am now active, click me...")

///////////////

// react to network events
$.ajax({
    url: "http://dynamic.xkcd.com/api-0/jsonp/comic?callback=?",
    dataType: "json",
    jsonpCallback: "xkcddata",
    success: function(data) {
        $("#xkcd-content").append(
            $("<h1/>").text(data.title),
            $("<img/>").attr({
                src: data.img,
                title: data.alt,
                alt: data.title
            })
        );
    }
});
